# 문제

[문제: https://acmicpc.net/problem/2740](https://boj.kr/2740)   
**난이도: 실버 V**

# 풀이 방법  

3중 반복문을 통해 행렬의 곱셈을 진행했다.   

# 소스 코드

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int A[101][101];
    int B[101][101];
    int ans[101][101];

    int n, m;
    cin >> n >> m;
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cin >> A[i][j];
        }
    }

    int k;
    cin >> m >> k;
    for(int i = 0; i < m; i++) {
        for(int j = 0; j < k; j++) {
            cin >> B[i][j];
        }
    }

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < k; j++) {
            for(int tmp = 0; tmp < m; tmp++) {
                ans[i][j] += (A[i][tmp] * B[tmp][j]);
            }
        }
    }

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < k; j++) {
            cout << ans[i][j] << " ";
        }
        cout << "\n";
    }

    return 0;
}
```
